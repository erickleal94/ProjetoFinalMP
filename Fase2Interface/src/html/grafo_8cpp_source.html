544

546  for (i = 1; i <= n - 1; i++) {
547  for (iterator1 = meu_grafo->vert; iterator1 != NULL; iterator1 = iterator1->next) {
548  if ((*dist)[iterator1->id] < 2e9) {//se a distancia nao for infinito
549  for (iterator2 = iterator1->sucessores; iterator2 != NULL; iterator2 = iterator2->next) {

552

555  (*dist)[iterator2->destino->id] = min + iterator2->peso;
556  }
557  }
